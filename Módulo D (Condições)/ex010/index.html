Condições Aninhadas

Condições Simples:

        
        if (cond 1) {
            bloco 1
        } else {
            if (cond 2) {
                bloco 2
            } else {
                bloco 3
            }
        }
        
        
        var idade = 12
        console.log(`Você tem ${idade} anos.`)
        if (idade < 16) {
            console.log(`Não vota.`)
        } else if (idade < 18 || idade > 65) {
            console.log(`Voto opcional.`)
        } else {
            console.log(`Voto obrigatório.`)
        }
        
        
        var agora = new Date()
        var hora = agora.getHours() //hora atual do pc
        console.log(`Agora são exatamente ${hora} horas.`)

        if (hora < 12) {
            console.log ('Bom dia!')
        } else if (hora <= 18) {
            console.log('Boa tarde!')
        } else {
            console.log('Boa noite!')
        }

Condições Múltiplas

        
        switch (expressão){
            case valor1:
                bloco 1
                break
            case valor2:
                bloco 2
                break
            case valor3:
                bloco 3
                break
            default:
                bloco default
                break
        }
