Funções

São ações executadas assim que são chamadas ou em decorrência de algum evento.

Uma função pode receber parâmetros e retornar um resultado.

        
            function ação(parâmetro 1) { -> parâmetro
                bloco 1 -> ação
                return res -> retorno
            }
            ação(parâmetro 2) -> chamada

            function parimp(n) {
                if (n % 2 == 0) {
                    return 'par'
                } else {
                    return 'impar'
                }
            }
            let res = parimp(11)
            console.log(res)

            function soma(n1=0, n2=0) {
                return n1 + n2
            }
            
            console.log(soma(2, 5))
            
            let v = function(x) {
                return x*2
            }
            
            console.log(v(5))
            
            function fatorial(n) {
                let fat = 1
                for (let c = n; c > 1; c--) {
                    fat *= c
                }
                return fat
            }
            
            console.log(fatorial(5))
            
            //RECURSIVIDADE
            function fatorial(n) {
                if (n == 1) {
                    return 1
                } else {
                    return n * fatorial(n-1)
                }
            }
            
            console.log(fatorial(4))
